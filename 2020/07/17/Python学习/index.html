程序三：字符串赋值

> string = "python";     //每个字符串中的字符对应一个索引，从0到5

> print(string[0:1000]); //结果为python，与String[0:6]结果相同

> print(string[-1:-1000:-1]);//结果为nohtyp ，反向取值。
> if "py" in string:      //结果为"正确",表示string这个字符串中含有py两个字符
>     print("正确");

文件的拷贝2

如果文件的内容过大，会占用内存，所以我们设置每次拷贝4个字节,使用循环完成所有数据的拷贝

file1 = open("C:\\Users\\hasee\\Desktop\\test.txt","rb");
file2 = open("C:\\Users\\hasee\\Desktop\\test2.txt","wb");
while True:#一直进行数据的拷贝，直到temp中的数据为空为止
    temp = file1.read(4*1024);
    if temp == b"":#判断数据是否为空，若为空，则表示数据所有数据拷贝完成
        break;
    file2.write(temp);

file1.close();
file2.close();

#面向对象
class Make_Plane(object):
    #定义属性
    #color = "green";
    #wheelNum = 4;
    #定义魔幻方法(相当于java中的构造方法)
    def __init__(self,color,wheelNum):#所有方法中形参self必不可少,同时，init前面是两个下划线，后面也是两个，总共四个
        self.color = color;#self相当于java中的this变量，在python中需要单独定义出来
        self.wheelNum = wheelNum;
    #定义普通方法
    def move(self):#注意在python中，self必不可少
        print("飞机起飞");
    def getPlaneInfo(self):
        print("轮胎数量%d,飞机颜色%s"%(self.wheelNum,self.color));

#创建对象
plane1 = Make_Plane("blue",3);
plane1.getPlaneInfo();

plane2= Make_Plane("green",20);
plane2.getPlaneInfo();

使用爬虫完成图片的下载，当我们想要批量下载一个网站的图片时，通过源码发现所有图片的名字都是进行以.jpg结尾。当我们使用浏览器下载图片时，在linux服务器的日志上显示的是浏览器的名称，如果使用python进行图片的下载，则日志会记录下python的详细信息，所以防火墙如果监测到，则会自动进行屏蔽。

class GetHtml(object):

    def GetList(self):

#使用正则表达式完成爬虫

完成 I say Good not food 过滤出good和food

import re
#1.单个字符的匹配
rs1 = re.findall(".ood","I say Good not food")
print(rs1)# 结果 ['Good', 'food'] ，其中点.的作用是代表任意单个字符

rs2 = re.findall("[Gf]ood","I say Good not food");
print(rs2) #结果为['Good', 'food']，其中【】的作用表示里面的字符会逐一与后面的字符进行匹配，

rs3 = re.findall("\d"," I am 23")
print(rs3) # 其中\d 表示任意单个数字

rs4 = re.findall("\w","I say Good not food and 88995");
print(rs4) #其中\w 表示0到9的数字和a到z的小写，以及A到Z的大写

rs5 = re.findall("\s","I say Good not food and  88995")
print(rs5) #其中\s 表示匹配任意的空白字符，空格,tab键

#2.字符串的匹配
rs6 = re.findall("Good","I say Good not food and 88995")
print(rs6) #这种是直接匹配

rs7 = re.findall("Good|food","I say Good not food and 88995")
print(rs7) #其中|字符的作用表示多个字符串进行逐个匹配

rs8 = re.findall("go*gle","I like google not ggle goooogle and gogle");
print(rs8) # 其中*号表示左邻字符出现0次或者多次

rs9 = re.findall("go+gle","I like google not ggle goooogle and gogle")
print(rs9) #其中+表示左邻字符出现1次或者多次

rs10 = re.findall("go?gle","I like google not ggle goooogle and gogle")
print(rs10) #？表示左邻字符出现0次或者一次

rs11 = re.findall("go{2,3}","I like google not ggle goooogle and gogle")
print(rs11) #{}表示定义左邻字符出现的次数

rs12 = re.findall("^I like","I like google not ggle goooogle and gogle")
print(rs12) #^表示是否以指定的字符开头

rs13 = re.findall("gogle$","I like google not ggle goooogle and gogle")
print(rs13)# $表示以指定的字符结尾

re14 = re.findall("(allen)\\1","my name is allenallen")
print(re14)
re14a = re.search("(allen)\\1","my name is allenallen")
print(re14a.group()) #()表示分组保存
